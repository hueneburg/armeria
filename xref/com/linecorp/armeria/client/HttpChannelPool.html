87      HttpChannelPool(HttpClientFactory clientFactory, EventLoop eventLoop,
88                      SslContext sslCtxHttp1Or2, SslContext sslCtxHttp1Only,
89                      ConnectionPoolListener listener) {
90          this.eventLoop = eventLoop;
91          pool = newEnumMap(
92                  Map.class,
93                  unused -> new HashMap<>(),
94                  SessionProtocol.H1, SessionProtocol.H1C,
95                  SessionProtocol.H2, SessionProtocol.H2C);
96          pendingAcquisitions = newEnumMap(
97                  Map.class,
98                  unused -> new HashMap<>(),
99                  SessionProtocol.HTTP, SessionProtocol.HTTPS,
100                 SessionProtocol.H1, SessionProtocol.H1C,
101                 SessionProtocol.H2, SessionProtocol.H2C);
102         allChannels = new IdentityHashMap<>();
103         this.listener = listener;
104 
105         final Bootstrap baseBootstrap = clientFactory.newBootstrap();
106         baseBootstrap.group(eventLoop);
107         bootstraps = newEnumMap(
108                 Bootstrap.class,
109                 desiredProtocol -> {
110                     final SslContext sslCtx = desiredProtocol == SessionProtocol.H1 ||
111                                               desiredProtocol == SessionProtocol.H1C ? sslCtxHttp1Only
112                                                                                      : sslCtxHttp1Or2;
113                     final Bootstrap bootstrap = baseBootstrap.clone();
114                     bootstrap.handler(new ChannelInitializer<Channel>() {
115                         @Override
116                         protected void initChannel(Channel ch) throws Exception {
117                             configureProxy(ch, clientFactory.proxyConfig(), sslCtx);
118                             ch.pipeline().addLast(
119                                     new HttpClientPipelineConfigurator(clientFactory, desiredProtocol, sslCtx));
120                         }
121                     });
122                     return bootstrap;
123                 },
124                 SessionProtocol.HTTP, SessionProtocol.HTTPS,
125                 SessionProtocol.H1, SessionProtocol.H1C,
126                 SessionProtocol.H2, SessionProtocol.H2C);
127         connectTimeoutMillis = (Integer) baseBootstrap.config().options()
128                                                       .get(ChannelOption.CONNECT_TIMEOUT_MILLIS);
129         useHttp1Pipelining = clientFactory.useHttp1Pipelining();
130     }

132     private void configureProxy(Channel ch, ProxyConfig proxyConfig, SslContext sslCtx) {
133         final ProxyHandler proxyHandler;
134         switch (proxyConfig.proxyType()) {
135             case DIRECT:
136                 return;
137             case SOCKS4:
138                 final Socks4ProxyConfig socks4ProxyConfig = (Socks4ProxyConfig) proxyConfig;
139                 proxyHandler = new Socks4ProxyHandler(socks4ProxyConfig.proxyAddress(),
140                                                       socks4ProxyConfig.username());
141                 break;
142             case SOCKS5:
143                 final Socks5ProxyConfig socks5ProxyConfig = (Socks5ProxyConfig) proxyConfig;
144                 proxyHandler = new Socks5ProxyHandler(
145                         socks5ProxyConfig.proxyAddress(), socks5ProxyConfig.username(),
146                         socks5ProxyConfig.password());
147                 break;
148             case CONNECT:
149                 final ConnectProxyConfig connectProxyConfig = (ConnectProxyConfig) proxyConfig;
150                 final String username = connectProxyConfig.username();
151                 final String password = connectProxyConfig.password();
152                 if (username == null || password == null) {
153                     proxyHandler = new HttpProxyHandler(connectProxyConfig.proxyAddress());
154                 } else {
155                     proxyHandler = new HttpProxyHandler(connectProxyConfig.proxyAddress(), username, password);
156                 }
157                 if (connectProxyConfig.useTls()) {
158                     ch.pipeline().addLast(sslCtx.newHandler(ch.alloc()));
159                 }
160                 break;
161             default:
162                 logger.warn("{} Ignoring unknown proxy type: {}", ch, proxyConfig.proxyType());
163                 return;
164         }
165         proxyHandler.setConnectTimeoutMillis(connectTimeoutMillis);
166         ch.pipeline().addLast(proxyHandler);
167     }

169     /**
170      * Returns an array whose index signifies {@link SessionProtocol#ordinal()}. Similar to {@link EnumMap}.
171      */
172     private static <T> T[] newEnumMap(Class<?> elementType,
173                                       Function<SessionProtocol, T> factory,
174                                       SessionProtocol... allowedProtocols) {
175         @SuppressWarnings("unchecked")
176         final T[] maps = (T[]) Array.newInstance(elementType, SessionProtocol.values().length);
177         // Attempting to access the array with an unallowed protocol will trigger NPE,
178         // which will help us find a bug.
179         for (SessionProtocol p : allowedProtocols) {
180             maps[p.ordinal()] = factory.apply(p);
181         }
182         return maps;
183     }

213     /**
214      * Attempts to acquire a {@link Channel} which is matched by the specified condition immediately.
215      *
216      * @return {@code null} is there's no match left in the pool and thus a new connection has to be
217      *         requested via {@link #acquireLater(SessionProtocol, PoolKey, ClientConnectionTimingsBuilder)}.
218      */
219     @Nullable
220     PooledChannel acquireNow(SessionProtocol desiredProtocol, PoolKey key) {
221         PooledChannel ch;
222         switch (desiredProtocol) {
223             case HTTP:
224                 ch = acquireNowExact(key, SessionProtocol.H2C);
225                 if (ch == null) {
226                     ch = acquireNowExact(key, SessionProtocol.H1C);
227                 }
228                 break;
229             case HTTPS:
230                 ch = acquireNowExact(key, SessionProtocol.H2);
231                 if (ch == null) {
232                     ch = acquireNowExact(key, SessionProtocol.H1);
233                 }
234                 break;
235             default:
236                 ch = acquireNowExact(key, desiredProtocol);
237         }
238         return ch;
239     }
240 
241     @Nullable
242     private PooledChannel acquireNowExact(PoolKey key, SessionProtocol protocol) {
243         final Deque<PooledChannel> queue = getPool(protocol, key);
244         if (queue == null) {
245             return null;
246         }
247 
248         // Find the most recently released channel while cleaning up the unhealthy channels.
249         for (int i = queue.size(); i > 0; i--) {
250             final PooledChannel pooledChannel = queue.peekLast();
251             assert pooledChannel != null;
252             if (!isHealthy(pooledChannel)) {
253                 queue.removeLast();
254                 continue;
255             }
256 
257             final HttpSession session = HttpSession.get(pooledChannel.get());
258             if (session.unfinishedResponses() >= session.maxUnfinishedResponses()) {
259                 // The channel is full of streams so we cannot create a new one.
260                 // Move the channel to the beginning of the queue so it has low priority.
261                 queue.removeLast();
262                 queue.addFirst(pooledChannel);
263                 continue;
264             }
265 
266             if (!protocol.isMultiplex()) {
267                 queue.removeLast();
268             }
269             return pooledChannel;
270         }
271 
272         return null;
273     }

275     private static boolean isHealthy(PooledChannel pooledChannel) {
276         final Channel ch = pooledChannel.get();
277         return ch.isActive() && HttpSession.get(ch).canSendRequest();
278     }
279 
280     @Nullable
281     private static SessionProtocol getProtocolIfHealthy(Channel ch) {
282         if (!ch.isActive()) {
283             return null;
284         }
285 
286         // Note that we do not need to check 'HttpSession.isActive()'
287         // because an inactive session always returns null.
288         return HttpSession.get(ch).protocol();
289     }

291     /**
292      * Acquires a new {@link Channel} which is matched by the specified condition by making a connection
293      * attempt or waiting for the current connection attempt in progress.
294      */
295     CompletableFuture<PooledChannel> acquireLater(SessionProtocol desiredProtocol, PoolKey key,
296                                                   ClientConnectionTimingsBuilder timingsBuilder) {
297         final CompletableFuture<PooledChannel> promise = new CompletableFuture<>();
298         if (!usePendingAcquisition(desiredProtocol, key, promise, timingsBuilder)) {
299             connect(desiredProtocol, key, promise, timingsBuilder);
300         }
301         return promise;
302     }

304     private CompletableFuture<PooledChannel> acquireLater(SessionProtocol desiredProtocol, PoolKey key,
305                                                           ClientConnectionTimingsBuilder timingsBuilder,
306                                                           CompletableFuture<PooledChannel> promise) {
307         if (!usePendingAcquisition(desiredProtocol, key, promise, timingsBuilder)) {
308             connect(desiredProtocol, key, promise, timingsBuilder);
309         }
310         return promise;
311     }

313     /**
314      * Tries to use the pending HTTP/2 connection to avoid creating an extra connection.
315      *
316      * @return {@code true} if succeeded to reuse the pending connection.
317      */
318     private boolean usePendingAcquisition(SessionProtocol desiredProtocol, PoolKey key,
319                                           CompletableFuture<PooledChannel> promise,
320                                           ClientConnectionTimingsBuilder timingsBuilder) {
321 
322         if (desiredProtocol == SessionProtocol.H1 || desiredProtocol == SessionProtocol.H1C) {
323             // Can't use HTTP/1 connections because they will not be available in the pool until
324             // the request is done.
325             return false;
326         }
327 
328         final CompletableFuture<PooledChannel> pendingAcquisition =
329                 getPendingAcquisition(desiredProtocol, key);
330 
331         if (pendingAcquisition == null) {
332             return false;
333         }
334 
335         timingsBuilder.pendingAcquisitionStart();
336         pendingAcquisition.handle((pch, cause) -> {
337             timingsBuilder.pendingAcquisitionEnd();
338 
339             if (cause == null) {
340                 final SessionProtocol actualProtocol = pch.protocol();
341                 if (actualProtocol.isMultiplex()) {
342                     final HttpSession session = HttpSession.get(pch.get());
343                     if (session.maxUnfinishedResponses() - session.unfinishedResponses() <= 1) {
344                         acquireLater(actualProtocol, key, timingsBuilder, promise);
345                     } else {
346                         promise.complete(pch);
347                     }
348                 } else {
349                     // Try to acquire again because the connection was not HTTP/2.
350                     // We use the exact protocol (H1 or H1C) instead of 'desiredProtocol' so that
351                     // we do not waste our time looking for pending acquisitions for the host
352                     // that does not support HTTP/2.
353                     final PooledChannel ch = acquireNow(actualProtocol, key);
354                     if (ch != null) {
355                         promise.complete(ch);
356                     } else {
357                         connect(actualProtocol, key, promise, timingsBuilder);
358                     }
359                 }
360             } else {
361                 // The pending connection attempt has failed.
362                 connect(desiredProtocol, key, promise, timingsBuilder);
363             }
364             return null;
365         });
366 
367         return true;
368     }

370     private void connect(SessionProtocol desiredProtocol, PoolKey key, CompletableFuture<PooledChannel> promise,
371                          ClientConnectionTimingsBuilder timingsBuilder) {
372 
373         setPendingAcquisition(desiredProtocol, key, promise);
374         timingsBuilder.socketConnectStart();
375 
376         final InetSocketAddress remoteAddress;
377         try {
378             remoteAddress = toRemoteAddress(key);
379         } catch (UnknownHostException e) {
380             notifyConnect(desiredProtocol, key, eventLoop.newFailedFuture(e), promise, timingsBuilder);
381             return;
382         }
383 
384         // Fail immediately if it is sure that the remote address doesn't support the desired protocol.
385         if (SessionProtocolNegotiationCache.isUnsupported(remoteAddress, desiredProtocol)) {
386             notifyConnect(desiredProtocol, key,
387                           eventLoop.newFailedFuture(
388                                   new SessionProtocolNegotiationException(
389                                           desiredProtocol, "previously failed negotiation")),
390                           promise, timingsBuilder);
391             return;
392         }
393 
394         // Create a new connection.
395         final Promise<Channel> sessionPromise = eventLoop.newPromise();
396         connect(remoteAddress, desiredProtocol, sessionPromise);
397 
398         if (sessionPromise.isDone()) {
399             notifyConnect(desiredProtocol, key, sessionPromise, promise, timingsBuilder);
400         } else {
401             sessionPromise.addListener((Future<Channel> future) -> {
402                 notifyConnect(desiredProtocol, key, future, promise, timingsBuilder);
403             });
404         }
405     }

407     /**
408      * A low-level operation that triggers a new connection attempt. Used only by:
409      * <ul>
410      *   <li>{@link #connect(SessionProtocol, PoolKey, CompletableFuture, ClientConnectionTimingsBuilder)} -
411      *       The pool has been exhausted.</li>
412      *   <li>{@link HttpSessionHandler} - HTTP/2 upgrade has failed.</li>
413      * </ul>
414      */
415     void connect(SocketAddress remoteAddress, SessionProtocol desiredProtocol,
416                  Promise<Channel> sessionPromise) {
417         final Bootstrap bootstrap = getBootstrap(desiredProtocol);
418         final ChannelFuture connectFuture = bootstrap.connect(remoteAddress);
419 
420         connectFuture.addListener((ChannelFuture future) -> {
421             if (future.isSuccess()) {
422                 initSession(desiredProtocol, future, sessionPromise);
423             } else {
424                 sessionPromise.setFailure(future.cause());
425             }
426         });
427     }
428 
429     private static InetSocketAddress toRemoteAddress(PoolKey key) throws UnknownHostException {
430         final InetAddress inetAddr = InetAddress.getByAddress(
431                 key.host, NetUtil.createByteArrayFromIpAddressString(key.ipAddr));
432         return new InetSocketAddress(inetAddr, key.port);
433     }

435     private void initSession(SessionProtocol desiredProtocol, ChannelFuture connectFuture,
436                              Promise<Channel> sessionPromise) {
437         assert connectFuture.isSuccess();
438 
439         final Channel ch = connectFuture.channel();
440         final EventLoop eventLoop = ch.eventLoop();
441         assert eventLoop.inEventLoop();
442 
443         final ScheduledFuture<?> timeoutFuture = eventLoop.schedule(() -> {
444             if (sessionPromise.tryFailure(new SessionProtocolNegotiationException(
445                     desiredProtocol, "connection established, but session creation timed out: " + ch))) {
446                 ch.close();
447             }
448         }, connectTimeoutMillis, TimeUnit.MILLISECONDS);
449 
450         ch.pipeline().addLast(new HttpSessionHandler(this, ch, sessionPromise, timeoutFuture,
451                                                      useHttp1Pipelining));
452     }

454     private void notifyConnect(SessionProtocol desiredProtocol, PoolKey key, Future<Channel> future,
455                                CompletableFuture<PooledChannel> promise,
456                                ClientConnectionTimingsBuilder timingsBuilder) {
457         assert future.isDone();
458         removePendingAcquisition(desiredProtocol, key);
459 
460         timingsBuilder.socketConnectEnd();
461         try {
462             if (future.isSuccess()) {
463                 final Channel channel = future.getNow();
464                 final SessionProtocol protocol = getProtocolIfHealthy(channel);
465                 if (protocol == null || closeable.isClosing()) {
466                     channel.close();
467                     promise.completeExceptionally(
468                             new UnprocessedRequestException(
469                                     new ClosedSessionException("acquired an unhealthy connection")));
470                     return;
471                 }
472 
473                 allChannels.put(channel, Boolean.TRUE);
474 
475                 try {
476                     listener.connectionOpen(protocol,
477                                             (InetSocketAddress) channel.remoteAddress(),
478                                             (InetSocketAddress) channel.localAddress(),
479                                             channel);
480                 } catch (Exception e) {
481                     if (logger.isWarnEnabled()) {
482                         logger.warn("{} Exception handling {}.connectionOpen()",
483                                     channel, listener.getClass().getName(), e);
484                     }
485                 }
486 
487                 final HttpSession session = HttpSession.get(channel);
488                 if (session.unfinishedResponses() < session.maxUnfinishedResponses()) {
489                     if (protocol.isMultiplex()) {
490                         final Http2PooledChannel pooledChannel = new Http2PooledChannel(channel, protocol);
491                         addToPool(protocol, key, pooledChannel);
492                         promise.complete(pooledChannel);
493                     } else {
494                         promise.complete(new Http1PooledChannel(channel, protocol, key));
495                     }
496                 } else {
497                     // Server set MAX_CONCURRENT_STREAMS to 0, which means we can't send anything.
498                     channel.close();
499                     promise.completeExceptionally(
500                             new UnprocessedRequestException(RefusedStreamException.get()));
501                 }
502 
503                 channel.closeFuture().addListener(f -> {
504                     allChannels.remove(channel);
505 
506                     // Clean up old unhealthy channels by iterating from the beginning of the queue.
507                     final Deque<PooledChannel> queue = getPool(protocol, key);
508                     if (queue != null) {
509                         for (;;) {
510                             final PooledChannel pooledChannel = queue.peekFirst();
511                             if (pooledChannel == null || isHealthy(pooledChannel)) {
512                                 break;
513                             }
514                             queue.removeFirst();
515                         }
516                     }
517 
518                     try {
519                         listener.connectionClosed(protocol,
520                                                   (InetSocketAddress) channel.remoteAddress(),
521                                                   (InetSocketAddress) channel.localAddress(),
522                                                   channel);
523                     } catch (Exception e) {
524                         if (logger.isWarnEnabled()) {
525                             logger.warn("{} Exception handling {}.connectionClosed()",
526                                         channel, listener.getClass().getName(), e);
527                         }
528                     }
529                 });
530             } else {
531                 promise.completeExceptionally(new UnprocessedRequestException(future.cause()));
532             }
533         } catch (Exception e) {
534             promise.completeExceptionally(new UnprocessedRequestException(e));
535         }
536     }

538     /**
539      * Adds a {@link Channel} to this pool.
540      */
541     private void addToPool(SessionProtocol actualProtocol, PoolKey key, PooledChannel pooledChannel) {
542         assert eventLoop.inEventLoop() : Thread.currentThread().getName();
543         getOrCreatePool(actualProtocol, key).addLast(pooledChannel);
544     }

546     @Override
547     public CompletableFuture<?> closeAsync() {
548         return closeable.closeAsync();
549     }
550 
551     private void closeAsync(CompletableFuture<?> future) {
552         if (!eventLoop.inEventLoop()) {
553             eventLoop.execute(() -> closeAsync(future));
554             return;
555         }
556 
557         // NB: Make a copy first, because close() will trigger the closeFuture listener
558         //     which mutates allChannels back, causing ConcurrentModificationException.
559         final Channel[] allChannels = this.allChannels.keySet().toArray(EMPTY_CHANNELS);
560         final int numAllChannels = allChannels.length;
561         if (numAllChannels == 0) {
562             future.complete(null);
563             return;
564         }
565 
566         // Complete the given future when all channels are closed.
567         final ChannelFutureListener listener = new ChannelFutureListener() {
568             private int numRemainingChannels = numAllChannels;
569 
570             @Override
571             public void operationComplete(ChannelFuture unused) throws Exception {
572                 if (--numRemainingChannels <= 0) {
573                     future.complete(null);
574                 }
575             }
576         };
577 
578         for (Channel ch : allChannels) {
579             ch.close().addListener(listener);
580         }
581     }

583     /**
584      * Closes all {@link Channel}s managed by this pool.
585      */
586     @Override
587     public void close() {
588         if (Thread.currentThread() instanceof NonBlocking) {
589             // Avoid blocking operation if we're in an event loop, because otherwise we might see a dead lock
590             // while waiting for the channels to be closed.
591             closeable.closeAsync();
592         } else {
593             closeable.close();
594         }
595     }

597     static final class PoolKey {
598         final String host;
599         final String ipAddr;
600         final int port;
601         final int hashCode;
602 
603         PoolKey(String host, String ipAddr, int port) {
604             this.host = host;
605             this.ipAddr = ipAddr;
606             this.port = port;
607             hashCode = (host.hashCode() * 31 + ipAddr.hashCode()) * 31 + port;
608         }
609 
610         @Override
611         public boolean equals(@Nullable Object o) {
612             if (this == o) {
613                 return true;
614             }
615 
616             if (!(o instanceof PoolKey)) {
617                 return false;
618             }
619 
620             final PoolKey that = (PoolKey) o;
621             // Compare IP address first, which is most likely to differ.
622             return ipAddr.equals(that.ipAddr) &&
623                    port == that.port &&
624                    host.equals(that.host);
625         }
626 
627         @Override
628         public int hashCode() {
629             return hashCode;
630         }
631 
632         @Override
633         public String toString() {
634             return MoreObjects.toStringHelper(this)
635                               .add("host", host)
636                               .add("ipAddr", ipAddr)
637                               .add("port", port)
638                               .toString();
639         }
640     }

643         Http2PooledChannel(Channel channel, SessionProtocol protocol) {
644             super(channel, protocol);

648         public void release() {

653     final class Http1PooledChannel extends PooledChannel {
654         private final PoolKey key;
655 
656         Http1PooledChannel(Channel channel, SessionProtocol protocol, PoolKey key) {
657             super(channel, protocol);
658             this.key = key;
659         }
660 
661         @Override
662         public void release() {
663             if (!eventLoop.inEventLoop()) {
664                 eventLoop.execute(this::doRelease);
665             } else {
666                 doRelease();
667             }
668         }
669 
670         private void doRelease() {
671             if (isHealthy(this)) {
672                 // Channel turns out to be healthy. Add it back to the pool.
673                 addToPool(protocol(), key, this);
674             } else {
675                 // Channel not healthy. Do not add it back to the pool.
676             }
677         }
