305     private static final Map<String, BuiltInProperty> keyToEnum;
306 
307     static final String WILDCARD_STR = "*";
308     static final String WILDCARD_REGEX = '\\' + WILDCARD_STR;
309     private static final Pattern PORT_443 = Pattern.compile(":0*443$");
310     private static final Pattern PORT_80 = Pattern.compile(":0*80$");
311     private static final BaseEncoding lowerCasedBase16 = BaseEncoding.base16().lowerCase();
312 
313     static {
314         final ImmutableMap.Builder<String, BuiltInProperty> builder = ImmutableMap.builder();
315         for (BuiltInProperty k : BuiltInProperty.values()) {
316             builder.put(k.key, k);
317         }
318         keyToEnum = builder.build();
319     }
320 
321     static List<BuiltInProperty> findByKeyPattern(String keyPattern) {
322         final Pattern pattern = Pattern.compile(
323                 ("\\Q" + keyPattern + "\\E").replaceAll(WILDCARD_REGEX, "\\\\E.*\\\\Q"));
324         return keyToEnum.entrySet().stream()
325                         .filter(e -> pattern.matcher(e.getKey()).matches())
326                         .map(Entry::getValue)
327                         .collect(toImmutableList());
328     }
329 
330     @Nullable
331     static BuiltInProperty findByKey(String key) {
332         return keyToEnum.entrySet().stream()
333                         .filter(e -> e.getKey().equals(key))
334                         .map(Entry::getValue)
335                         .findFirst().orElse(null);
336     }
337 
338     private static BaseEncoding lowerCasedBase16() {
339         return lowerCasedBase16;
340     }

342     private static String getAuthority(RequestLog log) {
343         final RequestContext ctx = log.context();
344         if (log.isAvailable(RequestLogProperty.REQUEST_HEADERS)) {
345             final String authority = getAuthority0(ctx, log.requestHeaders());
346             if (authority != null) {
347                 return authority;
348             }
349         }
350 
351         final HttpRequest origReq = ctx.request();
352         if (origReq != null) {
353             final String authority = getAuthority0(ctx, origReq.headers());
354             if (authority != null) {
355                 return authority;
356             }
357         }
358 
359         final String authority;
360         if (ctx instanceof ServiceRequestContext) {
361             final ServiceRequestContext sCtx = (ServiceRequestContext) ctx;
362             final int port = ((InetSocketAddress) sCtx.remoteAddress()).getPort();
363             final String hostname = sCtx.config().virtualHost().defaultHostname();
364             if (port == ctx.sessionProtocol().defaultPort()) {
365                 authority = hostname;
366             } else {
367                 authority = hostname + ':' + port;
368             }
369         } else {
370             final ClientRequestContext cCtx = (ClientRequestContext) ctx;
371             final Endpoint endpoint = cCtx.endpoint();
372             if (endpoint == null) {
373                 authority = "UNKNOWN";
374             } else {
375                 final int defaultPort = cCtx.sessionProtocol().defaultPort();
376                 final int port = endpoint.port(defaultPort);
377                 if (port == defaultPort) {
378                     authority = endpoint.host();
379                 } else {
380                     authority = endpoint.host() + ':' + port;
381                 }
382             }
383         }
384         return authority;
385     }
386 
387     @Nullable
388     private static String getAuthority0(RequestContext ctx, HttpHeaders headers) {
389         String authority = headers.get(HttpHeaderNames.AUTHORITY);
390         if (authority != null) {
391             final Pattern portPattern = ctx.sessionProtocol().isTls() ? PORT_443 : PORT_80;
392             final Matcher m = portPattern.matcher(authority);
393             if (m.find()) {
394                 authority = authority.substring(0, m.start());
395             }
396             return authority;
397         }
398 
399         return null;
400     }
