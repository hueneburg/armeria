43  /**
44   * Exports the specified properties from current {@link RequestContext} to {@link Map}.
45   */
46  public final class RequestContextExporter {
47  
48      @SuppressWarnings("rawtypes")
49      private static final ExportEntry[] EMPTY_EXPORT_ENTRIES = new ExportEntry[0];
50  
51      @VisibleForTesting
52      static final AttributeKey<State> STATE = AttributeKey.valueOf(RequestContextExporter.class, "STATE");
53  
54      /**
55       * Returns a newly created {@link RequestContextExporterBuilder}.
56       */
57      public static RequestContextExporterBuilder builder() {
58          return new RequestContextExporterBuilder();
59      }
60  
61      @Nullable
62      private final BuiltInProperties builtInProperties;
63      @Nullable
64      private final ExportEntry<BuiltInProperty>[] builtInPropertyArray;
65      @Nullable
66      private final ExportEntry<AttributeKey<?>>[] attrs;
67      private final int numAttrs;
68      @Nullable
69      private final ExportEntry<AsciiString>[] reqHeaders;
70      @Nullable
71      private final ExportEntry<AsciiString>[] resHeaders;
72  
73      RequestContextExporter(Set<ExportEntry<BuiltInProperty>> builtInPropertySet,
74                             Set<ExportEntry<AttributeKey<?>>> attrs,
75                             Set<ExportEntry<AsciiString>> reqHeaders,
76                             Set<ExportEntry<AsciiString>> resHeaders) {
77  
78          if (!builtInPropertySet.isEmpty()) {
79              builtInProperties = new BuiltInProperties();
80              builtInPropertyArray = builtInPropertySet.toArray(EMPTY_EXPORT_ENTRIES);
81              for (ExportEntry<BuiltInProperty> entry : builtInPropertyArray) {
82                  builtInProperties.add(entry.key);
83              }
84          } else {
85              builtInProperties = null;
86              builtInPropertyArray = null;
87          }
88  
89          if (!attrs.isEmpty()) {
90              this.attrs = attrs.toArray(EMPTY_EXPORT_ENTRIES);
91              numAttrs = this.attrs.length;
92          } else {
93              this.attrs = null;
94              numAttrs = 0;
95          }
96  
97          if (!reqHeaders.isEmpty()) {
98              this.reqHeaders = reqHeaders.toArray(EMPTY_EXPORT_ENTRIES);
99          } else {
100             this.reqHeaders = null;
101         }
102 
103         if (!resHeaders.isEmpty()) {
104             this.resHeaders = resHeaders.toArray(EMPTY_EXPORT_ENTRIES);
105         } else {
106             this.resHeaders = null;
107         }
108     }

110     /**
111      * Returns {@code true} if the specified {@link AttributeKey} is in the export list.
112      */
113     public boolean containsAttribute(AttributeKey<?> key) {
114         requireNonNull(key, "key");
115         if (attrs == null) {
116             return false;
117         }
118         return Arrays.stream(attrs).anyMatch(e -> e.key.equals(key));
119     }
120 
121     /**
122      * Returns {@code true} if the specified HTTP request header name is in the export list.
123      */
124     public boolean containsRequestHeader(CharSequence name) {
125         requireNonNull(name, "name");
126         if (reqHeaders == null) {
127             return false;
128         }
129         return Arrays.stream(reqHeaders).anyMatch(e -> e.key.contentEqualsIgnoreCase(name));
130     }
131 
132     /**
133      * Returns {@code true} if the specified HTTP response header name is in the export list.
134      */
135     public boolean containsResponseHeader(CharSequence name) {
136         requireNonNull(name, "name");
137         if (resHeaders == null) {
138             return false;
139         }
140         return Arrays.stream(resHeaders).anyMatch(e -> e.key.contentEqualsIgnoreCase(name));
141     }
142 
143     /**
144      * Returns {@code true} if the specified {@link BuiltInProperty} is in the export list.
145      */
146     public boolean containsBuiltIn(BuiltInProperty property) {
147         requireNonNull(property, "property");
148         if (builtInProperties == null) {
149             return false;
150         }
151         return builtInProperties.contains(property);
152     }
153 
154     /**
155      * Returns all {@link BuiltInProperty}s in the export list.
156      */
157     public Set<BuiltInProperty> builtIns() {
158         if (builtInPropertyArray == null) {
159             return ImmutableSet.of();
160         }
161         return Arrays.stream(builtInPropertyArray).map(entry -> entry.key).collect(toImmutableSet());
162     }
163 
164     /**
165      * Returns all {@link AttributeKey}s in the export list.
166      *
167      * @return the {@link Map} whose key is an alias and value is an {@link AttributeKey}
168      */
169     public Map<String, AttributeKey<?>> attributes() {
170         if (attrs == null) {
171             return ImmutableMap.of();
172         }
173         return Arrays.stream(attrs).collect(toImmutableMap(e -> e.exportKey, e -> e.key));
174     }
175 
176     /**
177      * Returns all HTTP request header names in the export list.
178      */
179     public Set<AsciiString> requestHeaders() {
180         if (reqHeaders == null) {
181             return ImmutableSet.of();
182         }
183         return Arrays.stream(reqHeaders).map(e -> e.key).collect(toImmutableSet());
184     }
185 
186     /**
187      * Returns all HTTP response header names in the export list.
188      */
189     public Set<AsciiString> responseHeaders() {
190         if (resHeaders == null) {
191             return ImmutableSet.of();
192         }
193         return Arrays.stream(resHeaders).map(e -> e.key).collect(toImmutableSet());
194     }
195 
196     /**
197      * Returns a {@link Map} whose key is an export key set through {@code add*()} in
198      * {@link RequestContextExporterBuilder} and value is extracted from {@link RequestContext}.
199      * Note that this method returns an empty {@link Map} if current {@link RequestContext} is {@code null}.
200      */
201     public Map<String, String> export() {
202         final RequestContext ctx = RequestContext.currentOrNull();
203         return ctx != null ? export(ctx) : ImmutableMap.of();
204     }
205 
206     /**
207      * Returns a {@link Map} whose key is an export key set through {@code add*()} in
208      * {@link RequestContextExporterBuilder} and value is extracted from the specified {@link RequestContext}.
209      */
210     public Map<String, String> export(RequestContext ctx) {
211         requireNonNull(ctx, "ctx");
212 
213         final State state = state(ctx);
214         final RequestLogAccess log = ctx.log();
215         boolean needsUpdate = false;
216 
217         // Needs to update if availabilityStamp has changed.
218         // Also updates `State.availabilityStamp` while checking.
219         final int availabilityStamp = log.availabilityStamp();
220         if (state.availabilityStamp != availabilityStamp) {
221             state.availabilityStamp = availabilityStamp;
222             needsUpdate = true;
223         }
224 
225         // Needs to update if any attributes have changed.
226         // Also updates `State.attrValues` while scanning.
227         if (attrs != null) {
228             assert state.attrValues != null;
229             for (int i = 0; i < attrs.length; i++) {
230                 final Object newValue = ctx.attr(attrs[i].key);
231                 if (!needsUpdate) {
232                     final Object oldValue = state.attrValues[i];
233                     needsUpdate = !Objects.equals(oldValue, newValue);
234                 }
235                 state.attrValues[i] = newValue;
236             }
237         }
238 
239         if (needsUpdate) {
240             export(state, ctx, log.partial());
241         }
242 
243         // Create a copy of 'state' to avoid the race between:
244         // - the delegate appenders who iterate over the MDC map and
245         // - this class who updates 'state'.
246         return state.clone();
247     }
248 
249     private void export(State state, RequestContext ctx, RequestLog log) {
250         exportBuiltIns(state, log);
251         exportAttributes(state);
252         exportRequestHeaders(state, log);
253         exportResponseHeaders(state, log);
254     }
255 
256     private void exportBuiltIns(State state, RequestLog log) {
257         if (builtInPropertyArray != null) {
258             for (final ExportEntry<BuiltInProperty> entry : builtInPropertyArray) {
259                 final String value = entry.key.converter.apply(log);
260                 if (value != null) {
261                     state.put(entry.exportKey, value);
262                 }
263             }
264         }
265     }
266 
267     private void exportAttributes(State state) {
268         if (attrs == null) {
269             return;
270         }
271 
272         assert state.attrValues != null;
273         for (int i = 0; i < numAttrs; i++) {
274             final ExportEntry<AttributeKey<?>> e = attrs[i];
275             putStringifiedProperty(state, e, state.attrValues[i]);
276         }
277     }
278 
279     private void exportRequestHeaders(State state, RequestLog log) {
280         if (reqHeaders == null || !log.isAvailable(RequestLogProperty.REQUEST_HEADERS)) {
281             return;
282         }
283 
284         exportHeaders(state, log.requestHeaders(), reqHeaders);
285     }
286 
287     private void exportResponseHeaders(State state, RequestLog log) {
288         if (resHeaders == null || !log.isAvailable(RequestLogProperty.RESPONSE_HEADERS)) {
289             return;
290         }
291 
292         exportHeaders(state, log.responseHeaders(), resHeaders);
293     }
294 
295     private static void exportHeaders(State state, HttpHeaders headers,
296                                       ExportEntry<AsciiString>[] requiredHeaderNames) {
297         for (ExportEntry<AsciiString> e : requiredHeaderNames) {
298             putStringifiedProperty(state, e, headers.get(e.key));
299         }
300     }
301 
302     private static void putStringifiedProperty(State state, ExportEntry<?> entry, @Nullable Object value) {
303         if (value != null) {
304             final String valueStr = entry.stringify(value);
305             if (valueStr != null) {
306                 state.put(entry.exportKey, valueStr);
307                 return;
308             }
309         }
310 
311         // Remove the value if it exists already.
312         state.remove(entry.exportKey);
313     }

315     static final class ExportEntry<T> {
316         final T key;
317         final String exportKey;
318         @Nullable
319         final Function<Object, String> stringifier;
320 
321         ExportEntry(T key, String exportKey) {
322             assert key != null;
323             assert exportKey != null;
324             this.key = key;
325             this.exportKey = exportKey;
326             stringifier = null;
327         }
328 
329         @SuppressWarnings("unchecked")
330         ExportEntry(T key, String exportKey, Function<?, ?> stringifier) {
331             assert key != null;
332             assert exportKey != null;
333             assert stringifier != null;
334             this.key = key;
335             this.exportKey = exportKey;
336             this.stringifier = (Function<Object, String>) stringifier;
337         }
338 
339         @Nullable
340         String stringify(@Nullable Object value) {
341             if (stringifier == null) {
342                 return value != null ? value.toString() : null;
343             } else {
344                 return stringifier.apply(value);
345             }
346         }
347 
348         @Override
349         public int hashCode() {
350             return key.hashCode() * 31 + exportKey.hashCode();
351         }
352 
353         @Override
354         public boolean equals(@Nullable Object o) {
355             if (this == o) {
356                 return true;
357             }
358 
359             if (!(o instanceof ExportEntry)) {
360                 return false;
361             }
362 
363             return key.equals(((ExportEntry<?>) o).key) &&
364                    exportKey.equals(((ExportEntry<?>) o).exportKey);
365         }
366 
367         @Override
368         public String toString() {
369             return exportKey + ':' + key;
370         }
371     }
372 
373     private State state(RequestContext ctx) {
374         final State state;
375         if (ctx instanceof ClientRequestContext) {
376             state = ((ClientRequestContext) ctx).ownAttr(STATE);
377         } else {
378             state = ctx.attr(STATE);
379         }
380 
381         if (state != null) {
382             return state;
383         }
384 
385         final State newState = new State(numAttrs);
386         ctx.setAttr(STATE, newState);
387         return newState;
388     }
389 
390     private static final class State extends Object2ObjectOpenHashMap<String, String> {
391         private static final long serialVersionUID = -7084248226635055988L;
392 
393         int availabilityStamp = -1;
394         @Nullable
395         final Object[] attrValues;
396 
397         State(int numAttrs) {
398             attrValues = numAttrs != 0 ? new Object[numAttrs] : null;
399         }
400     }
