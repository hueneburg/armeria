56  /**
57   * Provides information about a {@link Request}, its {@link Response} and related utilities.
58   * A server-side {@link Request} has a {@link ServiceRequestContext} and
59   * a client-side {@link Request} has a {@link ClientRequestContext}.
60   */
61  public interface RequestContext {
62  
63      /**
64       * Returns the context of the {@link Request} that is being handled in the current thread.
65       *
66       * @throws IllegalStateException if the context is unavailable in the current thread
67       */
68      static <T extends RequestContext> T current() {
69          final T ctx = currentOrNull();
70          if (ctx == null) {
71              throw new IllegalStateException(RequestContext.class.getSimpleName() + " unavailable");
72          }
73          return ctx;
74      }
75  
76      /**
77       * Returns the context of the {@link Request} that is being handled in the current thread.
78       *
79       * @return the {@link RequestContext} available in the current thread, or {@code null} if unavailable.
80       */
81      @Nullable
82      static <T extends RequestContext> T currentOrNull() {
83          return RequestContextUtil.get();
84      }
85  
86      /**
87       * Maps the context of the {@link Request} that is being handled in the current thread.
88       *
89       * @param mapper the {@link Function} that maps the {@link RequestContext}
90       * @param defaultValueSupplier the {@link Supplier} that provides the value when the context is unavailable
91       *                             in the current thread. If {@code null}, the {@code null} will be returned
92       *                             when the context is unavailable in the current thread.
93       */
94      @Nullable
95      static <T> T mapCurrent(
96              Function<? super RequestContext, T> mapper, @Nullable Supplier<T> defaultValueSupplier) {
97  
98          final RequestContext ctx = currentOrNull();
99          if (ctx != null) {
100             return mapper.apply(ctx);
101         }
102 
103         if (defaultValueSupplier != null) {
104             return defaultValueSupplier.get();
105         }
106 
107         return null;
108     }
109 
110     /**
111      * Returns the value mapped to the given {@link AttributeKey} or {@code null} if there's no value set by
112      * {@link #setAttr(AttributeKey, Object)} or {@link #setAttrIfAbsent(AttributeKey, Object)}.
113      */
114     @Nullable
115     <V> V attr(AttributeKey<V> key);
116 
117     /**
118      * Associates the specified value with the given {@link AttributeKey} in this context.
119      * If this context previously contained a mapping for the {@link AttributeKey},
120      * the old value is replaced by the specified value. Set {@code null} not to iterate the mapping from
121      * {@link #attrs()}.
122      */
123     <V> void setAttr(AttributeKey<V> key, @Nullable V value);
124 
125     /**
126      * Associates the specified value with the given {@link AttributeKey} in this context only
127      * if this context does not contain a mapping for the {@link AttributeKey}.
128      *
129      * @return {@code null} if there was no mapping for the {@link AttributeKey} or the old value if there's
130      *         a mapping for the {@link AttributeKey}.
131      */
132     @Nullable
133     <V> V setAttrIfAbsent(AttributeKey<V> key, V value);
134 
135     /**
136      * If the specified {@link AttributeKey} is not already associated with a value (or is mapped
137      * to {@code null}), attempts to compute its value using the given mapping
138      * function and stores it into this context.
139      *
140      * <p>If the mapping function returns {@code null}, no mapping is recorded.
141      *
142      * @return the current (existing or computed) value associated with
143      *         the specified {@link AttributeKey}, or {@code null} if the computed value is {@code null}
144      */
145     @Nullable
146     <V> V computeAttrIfAbsent(
147             AttributeKey<V> key, Function<? super AttributeKey<V>, ? extends V> mappingFunction);
148 
149     /**
150      * Returns the {@link Iterator} of all {@link Entry}s this context contains.
151      */
152     Iterator<Entry<AttributeKey<?>, Object>> attrs();

154     /**
155      * Returns the {@link HttpRequest} associated with this context, or {@code null} if there's no
156      * {@link HttpRequest} associated with this context yet.
157      */
158     @Nullable
159     HttpRequest request();

161     /**
162      * Returns the {@link RpcRequest} associated with this context, or {@code null} if there's no

166     RpcRequest rpcRequest();

168     /**
169      * Replaces the {@link HttpRequest} associated with this context with the specified one.
170      * This method is useful to a decorator that manipulates HTTP request headers.
171      *
172      * <p>Note that it is a bad idea to change the values of the pseudo headers ({@code ":method"},
173      * {@code ":path"}, {@code ":scheme"} and {@code ":authority"}) when replacing an {@link HttpRequest},
174      * because the properties of this context, such as {@link #path()}, are unaffected by such an attempt.</p>
175      *
176      * @see HttpRequest#withHeaders(RequestHeaders)
177      * @see HttpRequest#withHeaders(RequestHeadersBuilder)
178      */
179     void updateRequest(HttpRequest req);
180 
181     /**
182      * Replaces the {@link RpcRequest} associated with this context with the specified one.
183      * This method is useful to a decorator that manipulates an RPC call.
184      */
185     void updateRpcRequest(RpcRequest rpcReq);

256     /**
257      * Returns the {@link Executor} that is handling the current {@link Request}.
258      */
259     default Executor executor() {
260         // The implementation is the same as eventLoop but we expose as an Executor as well given
261         // how much easier it is to write tests for an Executor (i.e.,
262         // when(ctx.executor()).thenReturn(MoreExecutors.directExecutor()));
263         return eventLoop();
264     }
265 
266     /**
267      * Returns the {@link EventLoop} that is handling the current {@link Request}.
268      */
269     EventLoop eventLoop();
270 
271     /**
272      * Returns the {@link ByteBufAllocator} for this {@link RequestContext}. Any buffers created by this
273      * {@link ByteBufAllocator} must be
274      * <a href="https://netty.io/wiki/reference-counted-objects.html">reference-counted</a>. If you don't know
275      * what this means, you should probably use {@code byte[]} or {@link ByteBuffer} directly instead
276      * of calling this method.
277      */
278     default ByteBufAllocator alloc() {
279         throw new UnsupportedOperationException("No ByteBufAllocator available for this RequestContext.");
280     }
281 
282     /**
283      * Returns an {@link Executor} that will make sure this {@link RequestContext} is set as the current
284      * context before executing any callback. This should almost always be used for executing asynchronous
285      * callbacks in service code to make sure features that require the {@link RequestContext} work properly.
286      * Most asynchronous libraries like {@link CompletableFuture} provide methods that accept an
287      * {@link Executor} to run callbacks on.
288      */
289     default Executor contextAwareExecutor() {
290         // The implementation is the same as contextAwareEventLoop but we expose as an Executor as well given
291         // how common it is to use only as an Executor and it becomes much easier to write tests for an
292         // Executor (i.e., when(ctx.contextAwareExecutor()).thenReturn(MoreExecutors.directExecutor()));
293         return contextAwareEventLoop();
294     }
295 
296     /**
297      * Returns an {@link EventLoop} that will make sure this {@link RequestContext} is set as the current
298      * context before executing any callback.
299      */
300     default EventLoop contextAwareEventLoop() {
301         return new RequestContextAwareEventLoop(this, eventLoop());
302     }
303 
304     /**
305      * Pushes the specified context to the thread-local stack. To pop the context from the stack, call
306      * {@link SafeCloseable#close()}, which can be done using a {@code try-with-resources} block:
307      * <pre>{@code
308      * try (SafeCloseable ignored = ctx.push()) {
309      *     ...
310      * }
311      * }</pre>
312      *
313      * <p>This method may throw an {@link IllegalStateException} according to the status of the current
314      * thread-local. Please see {@link ServiceRequestContext#push()} and
315      * {@link ClientRequestContext#push()} to find out the satisfying conditions.
316      */
317     SafeCloseable push();
318 
319     /**
320      * Replaces the current {@link RequestContext} in the thread-local with this context without any validation.
321      * This method also does not run any callbacks.
322      *
323      * <p><strong>Note:</strong> Do not use this if you don't know what you are doing. This method does not
324      * prevent the situation where a wrong {@link RequestContext} is pushed into the thread-local.
325      * Use {@link #push()} instead.
326      *
327      * @see ClientRequestContext#push()
328      * @see ServiceRequestContext#push()
329      */
330     default SafeCloseable replace() {
331         final RequestContext oldCtx = RequestContextUtil.getAndSet(this);
332         return () -> RequestContextUtil.pop(this, oldCtx);
333     }
334 
335     /**
336      * Returns an {@link Executor} that will execute callbacks in the given {@code executor}, making sure to
337      * propagate the current {@link RequestContext} into the callback execution. It is generally preferred to
338      * use {@link #contextAwareEventLoop()} to ensure the callback stays on the same thread as well.
339      */
340     default Executor makeContextAware(Executor executor) {
341         return runnable -> executor.execute(makeContextAware(runnable));
342     }
343

345      * Returns an {@link ExecutorService} that will execute callbacks in the given {@code executor}, making

347      */
348     default ExecutorService makeContextAware(ExecutorService executor) {
349         return new RequestContextAwareExecutorService(this, executor);
350     }

360     /**
361      * Returns a {@link Callable} that makes sure the current {@link RequestContext} is set and then invokes
362      * the input {@code callable}.
363      */
364     default <T> Callable<T> makeContextAware(Callable<T> callable) {
365         return () -> {
366             try (SafeCloseable ignored = push()) {

368             }

370     }

432     /**
433      * Returns a {@link CompletionStage} that makes sure the current {@link RequestContext} is set and
434      * then invokes the input {@code stage}.
435      */
436     default <T> CompletionStage<T> makeContextAware(CompletionStage<T> stage) {
437         final CompletableFuture<T> future = JavaVersionSpecific.get().newRequestContextAwareFuture(this);
438         stage.handle((result, cause) -> {
439             try (SafeCloseable ignored = push()) {
440                 if (cause != null) {
441                     future.completeExceptionally(cause);
442                 } else {
443                     future.complete(result);
444                 }
445             } catch (Throwable t) {
446                 future.completeExceptionally(t);
447             }
448             return null;
449         });
450         return future;
451     }
452 
453     /**
454      * Returns a {@link CompletableFuture} that makes sure the current {@link RequestContext} is set and
455      * then invokes the input {@code future}.
456      */
457     default <T> CompletableFuture<T> makeContextAware(CompletableFuture<T> future) {
458         return makeContextAware((CompletionStage<T>) future).toCompletableFuture();
459     }
