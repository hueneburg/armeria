17  package com.linecorp.armeria.common;
18  
19  import javax.annotation.Nullable;
20  
21  import com.linecorp.armeria.common.util.UnstableApi;
22  
23  /**
24   * The storage for storing {@link RequestContext}.
25   *
26   * <p>If you want to implement your own storage or add some hooks when a {@link RequestContext} is pushed
27   * and popped, you should use {@link RequestContextStorageProvider}.
28   * Here's an example that sets MDC before {@link RequestContext} is pushed:
29   *
30   * <pre>{@code
31   * > public class MyStorage implements RequestContextStorageProvider {
32   * >
33   * >     @Override
34   * >     public RequestContextStorage newStorage() {
35   * >         RequestContextStorage threadLocalStorage = RequestContextStorage.threadLocal();
36   * >         return new RequestContextStorage() {
37   * >
38   * >             @Nullable
39   * >             @Override
40   * >             public <T extends RequestContext> T push(RequestContext toPush) {
41   * >                 setMdc(toPush);
42   * >                 return threadLocalStorage.push(toPush);
43   * >             }
44   * >
45   * >             @Override
46   * >             public void pop(RequestContext current, @Nullable RequestContext toRestore) {
47   * >                 clearMdc();
48   * >                 if (toRestore != null) {
49   * >                     setMdc(toRestore);
50   * >                 }
51   * >                 threadLocalStorage.pop(current, toRestore);
52   * >             }
53   * >             ...
54   * >          }
55   * >     }
56   * > }
57   * }</pre>
58   */
59  @UnstableApi
60  public interface RequestContextStorage {
61  
62      /**
63       * Returns the default {@link RequestContextStorage} which stores the {@link RequestContext}
64       * in the thread-local.
65       */
66      static RequestContextStorage threadLocal() {
67          return ThreadLocalRequestContextStorage.INSTANCE;
68      }
69  
70      /**
71       * Pushes the specified {@link RequestContext} into the storage.
72       *
73       * @return the old {@link RequestContext} which was in the storage before the specified {@code toPush} is
74       *         pushed. {@code null}, if there was no {@link RequestContext}.
75       */
76      @Nullable
77      <T extends RequestContext> T push(RequestContext toPush);
78  
79      /**
80       * Pops the current {@link RequestContext} in the storage and pushes back the specified {@code toRestore}.
81       * {@code toRestore} is the {@link RequestContext} returned from when
82       * {@linkplain #push(RequestContext) push(current)} is called, so it can be {@code null}.
83       *
84       * <p>The specified {@code current} must be the {@link RequestContext} in the storage. If it's not,
85       * it means that {@link RequestContext#push()} is not called using {@code try-with-resources} block, so
86       * the previous {@link RequestContext} is not popped properly.
87       */
88      void pop(RequestContext current, @Nullable RequestContext toRestore);
89  
90      /**
91       * Returns the {@link RequestContext} in the storage. {@code null} if there is no {@link RequestContext}.
92       */
93      @Nullable
94      <T extends RequestContext> T currentOrNull();
95  }
