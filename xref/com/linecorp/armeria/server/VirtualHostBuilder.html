194     /**
195      * Configures SSL or TLS of this {@link VirtualHost} with the specified {@code keyCertChainInputStream},
196      * {@code keyInputStream} and {@code keyPassword}.
197      *
198      * @see #tlsCustomizer(Consumer)
199      */
200     public VirtualHostBuilder tls(InputStream keyCertChainInputStream, InputStream keyInputStream,
201                                   @Nullable String keyPassword) {
202         requireNonNull(keyCertChainInputStream, "keyCertChainInputStream");
203         requireNonNull(keyInputStream, "keyInputStream");
204 
205         // Retrieve the content of the given streams so that they can be consumed more than once.
206         final byte[] keyCertChain;
207         final byte[] key;
208         try {
209             keyCertChain = ByteStreams.toByteArray(keyCertChainInputStream);
210             key = ByteStreams.toByteArray(keyInputStream);
211         } catch (IOException e) {
212             throw new IOError(e);
213         }
214 
215         return tls(() -> SslContextBuilder.forServer(new ByteArrayInputStream(keyCertChain),
216                                                      new ByteArrayInputStream(key),
217                                                      keyPassword));
218     }

377     /**
378      * Decorates and binds the specified {@link HttpServiceWithRoutes} at multiple {@link Route}s
379      * of the default {@link VirtualHost}.
380      *
381      * @param serviceWithRoutes the {@link HttpServiceWithRoutes}.
382      * @param decorators the decorator functions, which will be applied in the order specified.
383      */
384     public VirtualHostBuilder service(
385             HttpServiceWithRoutes serviceWithRoutes,
386             Iterable<? extends Function<? super HttpService, ? extends HttpService>> decorators) {
387         requireNonNull(serviceWithRoutes, "serviceWithRoutes");
388         requireNonNull(serviceWithRoutes.routes(), "serviceWithRoutes.routes()");
389         requireNonNull(decorators, "decorators");
390 
391         HttpService decorated = serviceWithRoutes;
392         for (Function<? super HttpService, ? extends HttpService> d : decorators) {
393             checkNotNull(d, "decorators contains null: %s", decorators);
394             decorated = d.apply(decorated);
395             checkNotNull(decorated, "A decorator returned null: %s", d);
396         }
397 
398         final HttpService finalDecorated = decorated;
399         serviceWithRoutes.routes().forEach(route -> service(route, finalDecorated));
400         return this;
401     }

582     @Nullable
583     private Function<? super HttpService, ? extends HttpService> getRouteDecoratingService(
584             @Nullable VirtualHostBuilder defaultVirtualHostBuilder) {
585         final List<RouteDecoratingService> routeDecoratingServices;
586         if (defaultVirtualHostBuilder != null) {
587             routeDecoratingServices = ImmutableList.<RouteDecoratingService>builder()
588                     .addAll(this.routeDecoratingServices)
589                     .addAll(defaultVirtualHostBuilder.routeDecoratingServices)
590                     .build();
591         } else {
592             routeDecoratingServices = ImmutableList.copyOf(this.routeDecoratingServices);
593         }
594 
595         if (!routeDecoratingServices.isEmpty()) {
596             return RouteDecoratingService.newDecorator(
597                     Routers.ofRouteDecoratingService(routeDecoratingServices));
598         } else {
599             return null;
600         }
601     }

807     /**
808      * Returns a newly-created {@link VirtualHost} based on the properties of this builder and the services
809      * added to this builder.
810      */
811     VirtualHost build(VirtualHostBuilder template) {
812         requireNonNull(template, "template");
813 
814         if (defaultHostname == null) {
815             if ("*".equals(hostnamePattern)) {
816                 defaultHostname = SystemInfo.hostname();
817             } else if (hostnamePattern.startsWith("*.")) {
818                 defaultHostname = hostnamePattern.substring(2);
819             } else {
820                 defaultHostname = hostnamePattern;
821             }
822         } else {
823             ensureHostnamePatternMatchesDefaultHostname(hostnamePattern, defaultHostname);
824         }
825 
826         // Retrieve all settings as a local copy. Use default builder's properties if not set.
827         final long requestTimeoutMillis =
828                 this.requestTimeoutMillis != null ?
829                 this.requestTimeoutMillis : template.requestTimeoutMillis;
830         final long maxRequestLength =
831                 this.maxRequestLength != null ?
832                 this.maxRequestLength : template.maxRequestLength;
833         final boolean verboseResponses =
834                 this.verboseResponses != null ?
835                 this.verboseResponses : template.verboseResponses;
836         final RejectedRouteHandler rejectedRouteHandler =
837                 this.rejectedRouteHandler != null ?
838                 this.rejectedRouteHandler : template.rejectedRouteHandler;
839 
840         final AccessLogWriter accessLogWriter;
841         final boolean shutdownAccessLogWriterOnStop;
842         if (this.accessLogWriter != null) {
843             accessLogWriter = this.accessLogWriter;
844             shutdownAccessLogWriterOnStop = this.shutdownAccessLogWriterOnStop;
845         } else {
846             accessLogWriter = template.accessLogWriter;
847             shutdownAccessLogWriterOnStop = template.shutdownAccessLogWriterOnStop;
848         }
849 
850         final Function<? super VirtualHost, ? extends Logger> accessLoggerMapper =
851                 this.accessLoggerMapper != null ?
852                 this.accessLoggerMapper : template.accessLoggerMapper;
853 
854         final AnnotatedServiceExtensions extensions =
855                 annotatedServiceExtensions != null ?
856                 annotatedServiceExtensions : template.annotatedServiceExtensions;
857 
858         assert rejectedRouteHandler != null;
859         assert accessLogWriter != null;
860         assert accessLoggerMapper != null;
861         assert extensions != null;
862 
863         final List<ServiceConfig> serviceConfigs = getServiceConfigSetters(template)
864                 .stream()
865                 .flatMap(cfgSetters -> {
866                     if (cfgSetters instanceof VirtualHostAnnotatedServiceBindingBuilder) {
867                         return ((VirtualHostAnnotatedServiceBindingBuilder) cfgSetters)
868                                 .buildServiceConfigBuilder(extensions).stream();
869                     } else if (cfgSetters instanceof AnnotatedServiceBindingBuilder) {
870                         return ((AnnotatedServiceBindingBuilder) cfgSetters)
871                                 .buildServiceConfigBuilder(extensions).stream();
872                     } else if (cfgSetters instanceof ServiceConfigBuilder) {
873                         return Stream.of((ServiceConfigBuilder) cfgSetters);
874                     } else {
875                         // Should not reach here.
876                         throw new Error("Unexpected service config setters type: " +
877                                         cfgSetters.getClass().getSimpleName());
878                     }
879                 }).map(cfgBuilder -> {
880                     return cfgBuilder.build(requestTimeoutMillis, maxRequestLength, verboseResponses,
881                                             accessLogWriter, shutdownAccessLogWriterOnStop);
882                 }).collect(toImmutableList());
883 
884         final ServiceConfig fallbackServiceConfig =
885                 new ServiceConfigBuilder(Route.ofCatchAll(), FallbackService.INSTANCE)
886                         .build(requestTimeoutMillis, maxRequestLength, verboseResponses,
887                                accessLogWriter, shutdownAccessLogWriterOnStop);
888 
889         SslContext sslContext = null;
890         boolean releaseSslContextOnFailure = false;
891         try {
892             // Whether the `SslContext` came (or was created) from this `VirtualHost`'s properties.
893             boolean sslContextFromThis = false;
894 
895             // Build a new SslContext or use a user-specified one for backward compatibility.
896             if (this.sslContext != null) {
897                 sslContext = this.sslContext;
898                 sslContextFromThis = true;
899             } else if (sslContextBuilderSupplier != null) {
900                 sslContext = buildSslContext(sslContextBuilderSupplier, tlsCustomizers);
901                 sslContextFromThis = true;
902                 releaseSslContextOnFailure = true;
903             } else if (template.sslContext != null) {
904                 sslContext = template.sslContext;
905             } else if (template.sslContextBuilderSupplier != null) {
906                 sslContext = buildSslContext(template.sslContextBuilderSupplier, template.tlsCustomizers);
907                 releaseSslContextOnFailure = true;
908             }
909 
910             // Generate a self-signed certificate if necessary.
911             if (sslContext == null) {
912                 final boolean tlsSelfSigned;
913                 final List<Consumer<? super SslContextBuilder>> tlsCustomizers;
914                 if (this.tlsSelfSigned != null) {
915                     tlsSelfSigned = this.tlsSelfSigned;
916                     tlsCustomizers = this.tlsCustomizers;
917                     sslContextFromThis = true;
918                 } else {
919                     tlsSelfSigned = template.tlsSelfSigned;
920                     tlsCustomizers = template.tlsCustomizers;
921                 }
922 
923                 if (tlsSelfSigned) {
924                     try {
925                         final SelfSignedCertificate ssc = selfSignedCertificate();
926                         sslContext = buildSslContext(() -> SslContextBuilder.forServer(ssc.certificate(),
927                                                                                        ssc.privateKey()),
928                                                      tlsCustomizers);
929                         releaseSslContextOnFailure = true;
930                     } catch (Exception e) {
931                         throw new RuntimeException("failed to create a self signed certificate", e);
932                     }
933                 }
934             }
935 
936             // Reject if a user called `tlsCustomizer()` without `tls()` or `tlsSelfSigned()`.
937             checkState(sslContextFromThis || tlsCustomizers.isEmpty(),
938                        "Cannot call tlsCustomizer() without tls() or tlsSelfSigned()");
939 
940             // Validate the built `SslContext`.
941             if (sslContext != null) {
942                 validateSslContext(sslContext);
943                 checkState(sslContext.isServer(), "sslContextBuilder built a client SSL context.");
944             }
945 
946             final VirtualHost virtualHost =
947                     new VirtualHost(defaultHostname, hostnamePattern, sslContext,
948                                     serviceConfigs, fallbackServiceConfig, rejectedRouteHandler,
949                                     accessLoggerMapper, requestTimeoutMillis, maxRequestLength,
950                                     verboseResponses, accessLogWriter, shutdownAccessLogWriterOnStop);
951 
952             final Function<? super HttpService, ? extends HttpService> decorator =
953                     getRouteDecoratingService(template);
954             final VirtualHost decoratedVirtualHost = decorator != null ? virtualHost.decorate(decorator)
955                                                                        : virtualHost;
956             releaseSslContextOnFailure = false;
957             return decoratedVirtualHost;
958         } finally {
959             if (releaseSslContextOnFailure) {
960                 ReferenceCountUtil.release(sslContext);
961             }
962         }
963     }
964 
965     private SelfSignedCertificate selfSignedCertificate() throws CertificateException {
966         if (selfSignedCertificate == null) {
967             return selfSignedCertificate = new SelfSignedCertificate(defaultHostname);
968         }
969         return selfSignedCertificate;
970     }

972     private static SslContext buildSslContext(
973             Supplier<SslContextBuilder> sslContextBuilderSupplier,
974             Iterable<? extends Consumer<? super SslContextBuilder>> tlsCustomizers) {
975         return SslContextUtil.createSslContext(sslContextBuilderSupplier, false, tlsCustomizers);

978     /**
979      * Makes sure the specified {@link SslContext} is configured properly. If configured as client context or
980      * key store password is not given to key store when {@link SslContext} was created using
981      * {@link KeyManagerFactory}, the validation will fail and an {@link IllegalStateException} will be raised.
982      */
983     private static SslContext validateSslContext(SslContext sslContext) {
984         if (!sslContext.isServer()) {
985             throw new IllegalArgumentException("sslContext: " + sslContext + " (expected: server context)");
986         }
987 
988         SSLEngine serverEngine = null;
989         SSLEngine clientEngine = null;
990 
991         try {
992             serverEngine = sslContext.newEngine(ByteBufAllocator.DEFAULT);
993             serverEngine.setUseClientMode(false);
994             serverEngine.setNeedClientAuth(false);
995 
996             final SslContext sslContextClient =
997                     buildSslContext(SslContextBuilder::forClient, ImmutableList.of());
998             clientEngine = sslContextClient.newEngine(ByteBufAllocator.DEFAULT);
999             clientEngine.setUseClientMode(true);
1000 
1001             final ByteBuffer appBuf = ByteBuffer.allocate(clientEngine.getSession().getApplicationBufferSize());
1002             final ByteBuffer packetBuf = ByteBuffer.allocate(clientEngine.getSession().getPacketBufferSize());
1003 
1004             clientEngine.wrap(appBuf, packetBuf);
1005             appBuf.clear();
1006             packetBuf.flip();
1007             serverEngine.unwrap(packetBuf, appBuf);
1008         } catch (SSLException e) {
1009             throw new IllegalStateException("failed to validate SSL/TLS configuration: " + e.getMessage(), e);
1010         } finally {
1011             ReferenceCountUtil.release(serverEngine);
1012             ReferenceCountUtil.release(clientEngine);
1013         }
1014 
1015         return sslContext;
1016     }
