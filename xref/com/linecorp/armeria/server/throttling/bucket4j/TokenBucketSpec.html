20  
21  import java.time.Duration;
22  import java.util.ArrayList;
23  import java.util.Arrays;
24  import java.util.Collections;

29  import javax.annotation.Nullable;

31  import com.google.common.base.Splitter;

33  /**
34   * A specification of a {@link TokenBucket} configuration represented by a string. The string syntax is
35   * a series of comma-separated {@link BandwidthLimit} configurations and each values is semicolon-separated,
36   * as per <a href="https://tools.ietf.org/id/draft-polli-ratelimit-headers-00.html">RateLimit Header Scheme for HTTP</a>.
37   *
38   * @see #parseTokenBucket(String) for detailed format of the specification.
39   */
40  final class TokenBucketSpec {
41  
42      private static final char OPTIONS_SEPARATOR = ';';
43      private static final char KEY_VALUE_SEPARATOR = '=';
44      private static final String PERIOD = "window";
45      private static final String OVERDRAFT = "burst";
46      private static final String INITIAL = "initial";
47      private static final Splitter.MapSplitter OPTIONS_SPLITTER =
48              Splitter.on(OPTIONS_SEPARATOR).trimResults().withKeyValueSeparator(KEY_VALUE_SEPARATOR);
49      private static final Splitter LIMITS_SPLITTER = Splitter.on(',').trimResults();
50  
51      private TokenBucketSpec() {}
52  
53      /**
54       * Creates a new {@link BandwidthLimit} that computes {@code limit}, {@code overdraftLimit},
55       * {@code initialSize} and {@code period} from a semicolon-separated {@code specification} string
56       * that conforms to the following format:
57       * <pre>{@code
58       * <limit>;window=<period(in seconds)>[;burst=<overdraftLimit>][;initial=<initialSize>]
59       * }</pre>
60       * All {@code specification} string elements must come in the defined order.
61       * For example:
62       * <ul>
63       *   <li>{@code 100;window=60;burst=1000} ({@code limit}=100, {@code overdraftLimit}=1000,
64       *       {@code initialSize} and {@code period}=60seconds)</li>
65       *   <li>{@code 100;window=60;burst=1000;initial=20} ({@code limit}=100, {@code overdraftLimit}=1000,
66       *       {@code initialSize}=20 and {@code period}=60seconds)</li>
67       *   <li>{@code 5000;window=1} ({@code limit}=5000 and {@code period}=1second)</li>
68       * </ul>
69       *
70       * @param specification the specification used to create a {@link BandwidthLimit}
71       */
72      static BandwidthLimit parseBandwidthLimit(String specification) {
73          requireNonNull(specification, "specification");
74          if (specification.isEmpty()) {
75              throw new IllegalArgumentException("Empty bandwidth limit specification");
76          }
77          final int limitSep = specification.indexOf(OPTIONS_SEPARATOR);
78          final long limit;
79          final Map<String, String> options;
80          if (limitSep > 0) {
81              limit = Long.parseLong(specification.substring(0, limitSep));
82              options = (limitSep < specification.length() - 1) ?
83                        OPTIONS_SPLITTER.split(specification.substring(limitSep + 1)) : Collections.emptyMap();
84          } else if (limitSep < 0) {
85              limit = Long.parseLong(specification);
86              options = Collections.emptyMap();
87          } else { // if (limitSep == 0)
88              throw new IllegalArgumentException("Invalid format of \"" +
89                                                 specification + "\" - limit not found");
90          }
91          if (!options.containsKey(PERIOD)) {
92              throw new IllegalArgumentException("Invalid format of \"" +
93                                                 specification + "\" - period not found");
94          }
95          final Duration period = Duration.ofSeconds(Long.parseLong(options.get(PERIOD)));
96          if (options.containsKey(OVERDRAFT)) {
97              final long overdraftLimit = Long.parseLong(options.get(OVERDRAFT));
98              if (options.containsKey(INITIAL)) {
99                  final long initialSize = Long.parseLong(options.get(INITIAL));
100                 return BandwidthLimit.of(limit, overdraftLimit, initialSize, period);
101             } else {
102                 return BandwidthLimit.of(limit, overdraftLimit, period);
103             }
104         } else {
105             return BandwidthLimit.of(limit, period);
106         }
107     }
108 
109     /**
110      * Creates a new {@link TokenBucket} from a comma-separated {@code specification} string
111      * that conforms to the following format:
112      * <pre>{@code
113      * <bandwidth limit 1>[, <bandwidth limit 2>[, etc.]]
114      * }</pre>
115      * The order of elements inside {@code specification} is not defined.
116      * For example:
117      * <ul>
118      *   <li>{@code 100;window=60;burst=1000, 50000;window=3600}</li>
119      * </ul>
120      *
121      * @param specification the specification used to create a {@link BandwidthLimit}.
122      *                      Empty {@link String} permitted to specify empty {@link TokenBucket}.
123      */
124     static TokenBucket parseTokenBucket(String specification) {
125         requireNonNull(specification, "specification");
126         final TokenBucketBuilder builder = TokenBucket.builder();
127         if (specification.isEmpty()) {
128             // empty specification allowed here
129             return builder.limits().build();
130         }
131         final List<BandwidthLimit> limits = new ArrayList<>(2);
132         for (String limitSpec: LIMITS_SPLITTER.split(specification)) {
133             limits.add(parseBandwidthLimit(limitSpec));
134         }
135         return builder.limits(limits).build();
136     }
137 
138     /**
139      * Returns a string representation of the {@link BandwidthLimit} in the following format:
140      * <pre>{@code
141      * <limit>;window=<period(in seconds)>[;burst=<overdraftLimit>][;policy="token bucket"]
142      * }</pre>
143      * For example: "100;window=60;burst=1000".
144      * <br>
145      * This method used to compose Quota Policy response header to inform the client about rate
146      * limiting policy as per <a href="https://tools.ietf.org/id/draft-polli-ratelimit-headers-00.html">RateLimit Header Scheme for HTTP</a>.
147      *
148      * @return A {@link String} representation of the {@link BandwidthLimit}.
149      */
150     @Nullable
151     static String toString(@Nullable BandwidthLimit bandwidthLimit) {
152         if (bandwidthLimit == null) {
153             return null;
154         }
155         final long limit = bandwidthLimit.limit();
156         final StringBuilder sb = new StringBuilder().append(limit);
157         sb.append(OPTIONS_SEPARATOR).append(PERIOD).append(KEY_VALUE_SEPARATOR)
158           .append(bandwidthLimit.period().getSeconds());
159         final long overdraftLimit = bandwidthLimit.overdraftLimit();
160         if (overdraftLimit > limit) {
161             sb.append(OPTIONS_SEPARATOR).append(OVERDRAFT).append(KEY_VALUE_SEPARATOR).append(overdraftLimit);
162         }
163         // We don't want to include INITIAL size with the Quota Policy response, since it's only relevant
164         // during the warm-up.
165         return sb.toString();
166     }
167 
168     /**
169      * Returns a string representation of the {@link TokenBucket} in the following format:
170      * <pre>{@code
171      * <first limit>;window=<first period(in seconds)>[;burst=<first overdraftLimit>],<SPACE>
172      * <second limit>;window=<second period(in seconds)>[;burst=<second overdraftLimit>],<SPACE>etc.
173      * }</pre>
174      * For example: "100;window=60;burst=1000, 5000;window=3600".
175      * <br>
176      * This method used to compose Quota Policy response header to inform the client about rate
177      * limiting policy as per <a href="https://tools.ietf.org/id/draft-polli-ratelimit-headers-00.html">RateLimit Header Scheme for HTTP</a>.
178      *
179      * @return A {@link String} representation of the {@link TokenBucket}.
180      */
181     @Nullable
182     static String toString(@Nullable TokenBucket tokenBucket) {
183         if (tokenBucket == null) {
184             return null;
185         }
186         final BandwidthLimit[] limits = tokenBucket.limits();
187         return Arrays.stream(limits).map(TokenBucketSpec::toString).collect(Collectors.joining(", "));
188     }
189 }
