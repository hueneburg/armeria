17  package com.linecorp.armeria.server.throttling.bucket4j;
18  
19  import java.time.Duration;
20  import java.util.concurrent.CompletableFuture;
21  import java.util.concurrent.CompletionStage;
22  import java.util.concurrent.TimeUnit;
23  
24  import javax.annotation.Nullable;
25  
26  import com.linecorp.armeria.common.HttpHeaderNames;
27  import com.linecorp.armeria.common.Request;
28  import com.linecorp.armeria.common.throttling.ThrottlingHeaders;
29  import com.linecorp.armeria.server.ServiceRequestContext;
30  import com.linecorp.armeria.server.throttling.ThrottlingStrategy;
31  
32  import io.github.bucket4j.AsyncBucket;
33  import io.github.bucket4j.Bucket4j;
34  import io.github.bucket4j.ConfigurationBuilder;
35  import io.github.bucket4j.local.LocalBucketBuilder;
36  
37  /**
38   * A {@link ThrottlingStrategy} that provides a throttling strategy based on Token-Bucket algorithm.
39   * The throttling works by examining the number of requests from the beginning, and
40   * throttling if the request rate exceed the configured bucket limits.
41   */
42  public final class TokenBucketThrottlingStrategy<T extends Request> extends ThrottlingStrategy<T> {
43  
44      /**
45       * Returns a newly created {@link TokenBucketThrottlingStrategyBuilder}.
46       */
47      public static <T extends Request> TokenBucketThrottlingStrategyBuilder<T> builder(TokenBucket tokenBucket) {
48          return new TokenBucketThrottlingStrategyBuilder<>(tokenBucket);
49      }
50  
51      private final AsyncBucket asyncBucket;
52      private final long minimumBackoffSeconds;
53      @Nullable
54      private final ThrottlingHeaders headersScheme;
55      @Nullable
56      private String quota;
57      private final boolean sendQuota;
58  
59      /**
60       * Creates a new named strategy with specified {@link TokenBucket} configuration,
61       * with minimum backoff period and with specific throttling headers scheme.
62       *
63       * @param tokenBucket {@link TokenBucket} configuration.
64       * @param minimumBackoff optional {@link Duration} that defines a minimum backoff period
65       *                       for throttled requests. By default, it will be set to 0 seconds.
66       * @param headersScheme optional {@link ThrottlingHeaders} to define specific RateLimit Header Scheme
67       *                      for HTTP. By default, no throttling headers will be used. The strategy will only use
68       *                      standard HTTP Retry-After header.
69       * @param sendQuota indicates whether to use quota header for the scheme.
70       * @param name optional name of the strategy. By default, it will be assigned with a predefined name.
71       */
72      TokenBucketThrottlingStrategy(TokenBucket tokenBucket,
73                                    @Nullable Duration minimumBackoff,
74                                    @Nullable ThrottlingHeaders headersScheme,
75                                    boolean sendQuota,
76                                    @Nullable String name) {
77          super(name);
78          // construct the bucket builder
79          final LocalBucketBuilder builder = Bucket4j.builder().withNanosecondPrecision();
80          for (BandwidthLimit limit : tokenBucket.limits()) {
81              builder.addLimit(limit.bandwidth());
82          }
83          // build the bucket
84          asyncBucket = builder.build().asAsync();
85          minimumBackoffSeconds = (minimumBackoff == null) ? 0L : minimumBackoff.getSeconds();
86          this.headersScheme = headersScheme;
87          this.sendQuota = sendQuota;
88          quota = sendQuota ? tokenBucket.toSpecString() : null;
89      }
90  
91      /**
92       * Resets the Token-Bucket configuration at runtime by providing a new set of limits.
93       * Empty set of limits will remove previously set limits. Reconfiguration will take place asynchronously.
94       * @param tokenBucket Token-Bucket configuration
95       * @return A {@link CompletableFuture} to handle asynchronous result
96       */
97      public CompletableFuture<Void> reconfigure(TokenBucket tokenBucket) {
98          // construct the configuration builder
99          final ConfigurationBuilder builder = Bucket4j.configurationBuilder();
100         for (BandwidthLimit limit : tokenBucket.limits()) {
101             builder.addLimit(limit.bandwidth());
102         }
103         // reconfigure the bucket
104         return asyncBucket.replaceConfiguration(builder.build())
105                           .thenRun(() -> quota = sendQuota ? tokenBucket.toSpecString() : null);
106     }
107 
108     /**
109      * Registers a request with the bucket.
110      */
111     @Override
112     public CompletionStage<Boolean> accept(final ServiceRequestContext ctx, final T request) {
113         return asyncBucket.tryConsumeAndReturnRemaining(1L).thenApply(probe -> {
114             final boolean accepted = probe.isConsumed();
115             final long remainingTokens = probe.getRemainingTokens();
116             final long remainingSeconds =
117                     TimeUnit.SECONDS.convert(probe.getNanosToWaitForRefill(), TimeUnit.NANOSECONDS);
118             // calculate maximum between pre-configured minimum backoff and remaining seconds
119             final long retryAfter = Math.max(minimumBackoffSeconds, remainingSeconds);
120             if (!accepted) {
121                 // always send Retry-After header for rejected requests
122                 ctx.addAdditionalResponseHeader(HttpHeaderNames.RETRY_AFTER, retryAfter);
123             }
124             if (headersScheme != null) {
125                 // when headers scheme defined,
126                 // add those to the response (either for accepted or rejected requests)
127                 ctx.addAdditionalResponseHeader(headersScheme.remainingHeader(), remainingTokens);
128                 ctx.addAdditionalResponseHeader(headersScheme.resetHeader(),
129                                                 accepted ? remainingSeconds : retryAfter);
130                 if (sendQuota && quota != null) {
131                     ctx.addAdditionalResponseHeader(headersScheme.limitHeader(), quota);
132                 }
133             }
134             return accepted;
135         });
136     }
137 }
