50  /**
51   * This will send an {@link Http2PingFrame} when an {@link IdleStateEvent} is emitted by {@link
52   * IdleStateHandler} and {@link Flags#defaultHttp2PingTimeoutMillis()} is greater that zero.
53   *
54   * <p>Once an {@link IdleStateEvent} is triggered and when there are active streams open then a
55   * {@link Http2PingFrame} will be written on connection. When there are no active streams then it depends on
56   * {@link Flags#defaultUseHttp2PingWhenNoActiveStreams()}.
57   *
58   * <p>Once a {@link Http2PingFrame} is written, then either an ACK for the {@link Http2PingFrame} or any data
59   * is read on connection will invalidate the condition that triggers connection closure. If either of the
60   * conditions are not met then the connection will be closed.
61   *
62   * <p>This class is <b>not</b> thread-safe and all methods are to be called from single thread such
63   * as {@link EventLoop}.
64   *
65   * @see Flags#defaultUseHttp2PingWhenNoActiveStreams()
66   * @see Flags#defaultHttp2PingTimeoutMillis()
67   */
68  @NotThreadSafe
69  public class Http2KeepAliveHandler {
70  
71      private static final Logger logger = LoggerFactory.getLogger(Http2KeepAliveHandler.class);
72  
73      @Nullable
74      private final Stopwatch stopwatch = logger.isDebugEnabled() ? Stopwatch.createUnstarted() : null;
75      private final boolean sendPingsOnNoActiveStreams;
76      private final long pingTimeoutMillis;
77      private final Http2FrameWriter frameWriter;
78      private final ThreadLocalRandom random = ThreadLocalRandom.current();
79      private final Http2Connection http2Connection;
80      private final Channel channel;
81      private final ChannelFutureListener pingWriteListener = new PingWriteListener();
82      private final Runnable shutdownRunnable = this::closeChannelAndLog;
83  
84      @Nullable
85      private ChannelFuture pingWriteFuture;
86      @Nullable
87      private Future<?> shutdownFuture;
88      private long lastPingPayload;
89      private State state = State.IDLE;
90  
91      public Http2KeepAliveHandler(Channel channel, Http2FrameWriter frameWriter, Http2Connection http2Connection,
92                                   long pingTimeoutMillis, boolean sendPingsOnNoActiveStreams) {
93          checkArgument(pingTimeoutMillis > 0, pingTimeoutMillis);
94          this.channel = requireNonNull(channel, "channel");
95          this.frameWriter = requireNonNull(frameWriter, "frameWriter");
96          this.pingTimeoutMillis = pingTimeoutMillis;
97          this.http2Connection = requireNonNull(http2Connection, "http2Connection");
98          this.sendPingsOnNoActiveStreams = sendPingsOnNoActiveStreams;
99      }
100 
101     public void onChannelIdle(ChannelHandlerContext ctx, IdleStateEvent event) {
102         logger.debug("{} {} triggered.", channel, event);
103 
104         if (!canSendPing()) {
105             // The default behaviour is to shutdown the channel on idle timeout if not HTTP/2 conn.
106             // So preserving the behaviour.
107             closeChannelAndLog();
108             return;
109         }
110 
111         // Only interested in ALL_IDLE event and when Http2KeepAliveHandler is ready.
112         // Http2KeepAliveHandler may not be ready because it is currently handling
113         // sending a PING or expecting a PING ACK on channel.
114         if (state != State.IDLE || event.state() != IdleState.ALL_IDLE) {
115             return;
116         }
117 
118         writePing(ctx);
119     }
120 
121     private boolean canSendPing() {
122         if (http2Connection.numActiveStreams() == 0) {
123             return sendPingsOnNoActiveStreams;
124         } else {
125             return true;
126         }
127     }
128 
129     private void writePing(ChannelHandlerContext ctx) {
130         lastPingPayload = random.nextLong();
131         state = State.PING_SCHEDULED;
132         pingWriteFuture = frameWriter.writePing(ctx, false, lastPingPayload, ctx.newPromise())
133                                      .addListener(pingWriteListener);
134         ctx.flush();
135     }
136 
137     /**
138      * Callback for when channel is in-active to cleans up resources.
139      */
140     public void onChannelInactive() {
141         state = State.SHUTDOWN;
142         cancelFutures();
143     }
144 
145     public void onChannelRead() {
146         state = State.IDLE;
147         cancelFutures();
148     }
149 
150     private void resetFutures() {
151         shutdownFuture = null;
152         pingWriteFuture = null;
153     }
154 
155     private void cancelFutures() {
156         if (shutdownFuture != null) {
157             shutdownFuture.cancel(false);
158             shutdownFuture = null;
159         }
160         if (pingWriteFuture != null) {
161             pingWriteFuture.cancel(false);
162             pingWriteFuture = null;
163         }
164     }
165 
166     public void onPingAck(long data) {
167         final long elapsed = getStopwatchElapsedInNanos();
168         if (!isGoodPingAck(data)) {
169             return;
170         }
171 
172         if (shutdownFuture != null) {
173             final boolean isCancelled = shutdownFuture.cancel(false);
174             if (!isCancelled) {
175                 logger.debug("{} shutdownFuture cannot be cancelled because of late PING ACK", channel);
176             }
177         }
178         logger.debug("{} PING(ACK=1, DATA={}) received in {} ns", channel, lastPingPayload, elapsed);
179         state = State.IDLE;
180         resetFutures();
181     }
182 
183     private boolean isGoodPingAck(long data) {
184         // This condition can be true when channel read some data other than PING ACK frame
185         // or a PING ACK is received without sending PING in first place.
186         if (state != State.PENDING_PING_ACK) {
187             logger.debug("{} PING(ACK=1, DATA={}) ignored", channel, data);
188             return false;
189         }
190         if (lastPingPayload != data) {
191             logger.debug("{} Unexpected PING(ACK=1, DATA={}) received, " +
192                          "but expecting PING(ACK=1, DATA={})", channel, data, lastPingPayload);
193             return false;
194         }
195         return true;
196     }
197 
198     @VisibleForTesting
199     State state() {
200         return state;
201     }
202 
203     @VisibleForTesting
204     long lastPingPayload() {
205         return lastPingPayload;
206     }
207 
208     private void closeChannelAndLog() {
209         if (state == State.SHUTDOWN) {
210             return;
211         }
212         logger.debug("{} Closing an idle channel", channel);
213         channel.close().addListener(future -> {
214             if (future.isSuccess()) {
215                 logger.debug("{} Closed an idle channel", channel);
216             } else {
217                 logger.debug("{} Failed to close an idle channel", channel, future.cause());
218             }
219             state = State.SHUTDOWN;
220         });
221     }
222 
223     private long getStopwatchElapsedInNanos() {
224         if (stopwatch == null) {
225             return -1;
226         }
227         return stopwatch.elapsed(TimeUnit.NANOSECONDS);
228     }
229 
230     /**
231      * State changes from IDLE -> PING_SCHEDULED -> PENDING_PING_ACK -> IDLE and so on. When the
232      * channel is inactive then the state changes to SHUTDOWN.
233      */
234     enum State {
235         /* Nothing happening, but waiting for IdleStateEvent */
236         IDLE,
237 
238         /* PING is scheduled */
239         PING_SCHEDULED,
240 
241         /* PING is sent and is pending ACK */
242         PENDING_PING_ACK,
243 
244         /* Not active anymore */
245         SHUTDOWN
246     }
247 
248     private class PingWriteListener implements ChannelFutureListener {
249 
250         @Override
251         public void operationComplete(ChannelFuture future) throws Exception {
252             if (future.isSuccess()) {
253                 logger.debug("{} PING(ACK=0, DATA={}) write successful", channel, lastPingPayload);
254                 final EventLoop el = channel.eventLoop();
255                 shutdownFuture = el.schedule(shutdownRunnable, pingTimeoutMillis, TimeUnit.MILLISECONDS);
256                 state = State.PENDING_PING_ACK;
257                 resetStopwatch();
258             } else {
259                 // Mostly because the channel is already closed. So ignore and change state to IDLE.
260                 // If the channel is closed, we change state to SHUTDOWN on onChannelInactive.
261                 if (!future.isCancelled() && Exceptions.isExpected(future.cause())) {
262                     logger.debug("{} PING write failed", channel, future.cause());
263                 }
264                 if (state != State.SHUTDOWN) {
265                     state = State.IDLE;
266                 }
267             }
268         }
269 
270         private void resetStopwatch() {
271             if (stopwatch != null) {
272                 stopwatch.reset().start();
273             }
274         }
275     }
