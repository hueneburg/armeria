49  public abstract class Http1ObjectEncoder implements HttpObjectEncoder {
50  
51      /**
52       * The maximum allowed length of an HTTP chunk when TLS is enabled.
53       * <ul>
54       *   <li>16384 - The maximum length of a cleartext TLS record.</li>
55       *   <li>6 - The maximum header length of an HTTP chunk. i.e. "4000\r\n".length()</li>
56       * </ul>
57       *
58       * <p>To be precise, we have a chance of wasting 6 bytes because we may not use chunked encoding,
59       * but it is not worth adding complexity to be that precise.
60       */
61      private static final int MAX_TLS_DATA_LENGTH = 16384 - 6;
62  
63      /**
64       * A non-last empty {@link HttpContent}.
65       */
66      private static final HttpContent EMPTY_CONTENT = new DefaultHttpContent(Unpooled.EMPTY_BUFFER);
67  
68      private final Channel ch;
69      private final SessionProtocol protocol;
70  
71      private volatile boolean closed;
72  
73      /**
74       * The ID of the request which is at its turn to send a response.
75       */
76      private int currentId = 1;
77  
78      /**
79       * The minimum ID of the request whose stream has been closed/reset.
80       */
81      private int minClosedId = Integer.MAX_VALUE;
82  
83      /**
84       * The maximum known ID with pending writes.
85       */
86      private int maxIdWithPendingWrites = Integer.MIN_VALUE;
87  
88      /**
89       * The map which maps a request ID to its related pending response.
90       */
91      private final IntObjectMap<PendingWrites> pendingWritesMap = new IntObjectHashMap<>();
92  
93      protected Http1ObjectEncoder(Channel ch, SessionProtocol protocol) {
94          this.ch = requireNonNull(ch, "ch");
95          this.protocol = requireNonNull(protocol, "protocol");
96      }

98      @Override
99      public final Channel channel() {
100         return ch;
101     }

103     protected final ChannelFuture writeNonInformationalHeaders(int id, HttpObject converted,
104                                                                boolean endStream) {
105         ChannelFuture f;
106         if (converted instanceof LastHttpContent) {
107             assert endStream;
108             f = write(id, converted, true);
109         } else {
110             f = write(id, converted, false);
111             if (endStream) {
112                 f = write(id, LastHttpContent.EMPTY_LAST_CONTENT, true);
113             }
114         }
115         ch.flush();
116         return f;
117     }
118 
119     @Override
120     public final ChannelFuture doWriteData(int id, int streamId, HttpData data, boolean endStream) {
121         if (!isWritable(id)) {
122             ReferenceCountUtil.safeRelease(data);
123             return newClosedSessionFuture();
124         }
125 
126         final int length = data.length();
127         if (length == 0) {
128             ReferenceCountUtil.safeRelease(data);
129             final HttpContent content = endStream ? LastHttpContent.EMPTY_LAST_CONTENT : EMPTY_CONTENT;
130             final ChannelFuture future = write(id, content, endStream);
131             ch.flush();
132             return future;
133         }
134 
135         try {
136             if (!protocol.isTls() || length <= MAX_TLS_DATA_LENGTH) {
137                 // Cleartext connection or data.length() <= MAX_TLS_DATA_LENGTH
138                 return doWriteUnsplitData(id, data, endStream);
139             } else {
140                 // TLS or data.length() > MAX_TLS_DATA_LENGTH
141                 return doWriteSplitData(id, data, endStream);
142             }
143         } catch (Throwable t) {
144             return newFailedFuture(t);
145         }
146     }
147 
148     private ChannelFuture doWriteUnsplitData(int id, HttpData data, boolean endStream) {
149         final ByteBuf buf = toByteBuf(data);
150         boolean handled = false;
151         try {
152             final HttpContent content;
153             if (endStream) {
154                 content = new DefaultLastHttpContent(buf);
155             } else {
156                 content = new DefaultHttpContent(buf);
157             }
158 
159             final ChannelFuture future = write(id, content, endStream);
160             handled = true;
161             ch.flush();
162             return future;
163         } finally {
164             if (!handled) {
165                 ReferenceCountUtil.safeRelease(buf);
166             }
167         }
168     }
169 
170     private ChannelFuture doWriteSplitData(int id, HttpData data, boolean endStream) {
171         try {
172             int offset = 0;
173             int remaining = data.length();
174             ChannelFuture lastFuture;
175             for (;;) {
176                 // Ensure an HttpContent does not exceed the maximum length of a cleartext TLS record.
177                 final int chunkSize = Math.min(MAX_TLS_DATA_LENGTH, remaining);
178                 lastFuture = write(id, new DefaultHttpContent(dataChunk(data, offset, chunkSize)), false);
179                 remaining -= chunkSize;
180                 if (remaining == 0) {
181                     break;
182                 }
183                 offset += chunkSize;
184             }
185 
186             if (endStream) {
187                 lastFuture = write(id, LastHttpContent.EMPTY_LAST_CONTENT, true);
188             }
189 
190             ch.flush();
191             return lastFuture;
192         } finally {
193             ReferenceCountUtil.safeRelease(data);
194         }
195     }
196 
197     private static ByteBuf dataChunk(HttpData data, int offset, int chunkSize) {
198         if (data instanceof ByteBufHolder) {
199             final ByteBuf buf = ((ByteBufHolder) data).content();
200             return buf.retainedSlice(offset, chunkSize);
201         } else {
202             return Unpooled.wrappedBuffer(data.array(), offset, chunkSize);
203         }
204     }
205 
206     protected final ChannelFuture write(int id, HttpObject obj, boolean endStream) {
207         if (id < currentId) {
208             // Attempted to write something on a finished request/response; discard.
209             // e.g. the request already timed out.
210             ReferenceCountUtil.safeRelease(obj);
211             return newFailedFuture(ClosedStreamException.get());
212         }
213 
214         final PendingWrites currentPendingWrites = pendingWritesMap.get(id);
215         if (id == currentId) {
216             if (currentPendingWrites != null) {
217                 pendingWritesMap.remove(id);
218                 flushPendingWrites(currentPendingWrites);
219             }
220 
221             final ChannelFuture future = ch.write(obj);
222             if (endStream) {
223                 currentId++;
224 
225                 // The next PendingWrites might be complete already.
226                 for (;;) {
227                     final PendingWrites nextPendingWrites = pendingWritesMap.get(currentId);
228                     if (nextPendingWrites == null) {
229                         break;
230                     }
231 
232                     flushPendingWrites(nextPendingWrites);
233                     if (!nextPendingWrites.isEndOfStream()) {
234                         break;
235                     }
236 
237                     pendingWritesMap.remove(currentId);
238                     currentId++;
239                 }
240             }
241 
242             return future;
243         } else {
244             final ChannelPromise promise = ch.newPromise();
245             final Entry<HttpObject, ChannelPromise> entry = new SimpleImmutableEntry<>(obj, promise);
246             final PendingWrites pendingWrites;
247             if (currentPendingWrites == null) {
248                 pendingWrites = new PendingWrites();
249                 maxIdWithPendingWrites = Math.max(maxIdWithPendingWrites, id);
250                 pendingWritesMap.put(id, pendingWrites);
251             } else {
252                 pendingWrites = currentPendingWrites;
253             }
254 
255             pendingWrites.add(entry);
256 
257             if (endStream) {
258                 pendingWrites.setEndOfStream();
259             }
260 
261             return promise;
262         }
263     }
264 
265     private void flushPendingWrites(PendingWrites pendingWrites) {
266         for (;;) {
267             final Entry<HttpObject, ChannelPromise> e = pendingWrites.poll();
268             if (e == null) {
269                 break;
270             }
271 
272             ch.write(e.getKey(), e.getValue());
273         }
274     }

276     @Override
277     public ChannelFuture doWriteTrailers(int id, int streamId, HttpHeaders headers) {
278         if (!isWritable(id)) {
279             return newClosedSessionFuture();
280         }
281 
282         return write(id, convertTrailers(headers), true);
283     }
284 
285     private LastHttpContent convertTrailers(HttpHeaders inputHeaders) {
286         if (inputHeaders.isEmpty()) {
287             return LastHttpContent.EMPTY_LAST_CONTENT;
288         }
289 
290         final LastHttpContent lastContent = new DefaultLastHttpContent(Unpooled.EMPTY_BUFFER, false);
291         final io.netty.handler.codec.http.HttpHeaders outputHeaders = lastContent.trailingHeaders();
292         convertTrailers(inputHeaders, outputHeaders);
293         return lastContent;
294     }
295 
296     protected abstract void convertTrailers(HttpHeaders inputHeaders,
297                                             io.netty.handler.codec.http.HttpHeaders outputHeaders);

299     @Override
300     public final ChannelFuture doWriteReset(int id, int streamId, Http2Error error) {
301         // NB: this.minClosedId can be overwritten more than once when 3+ pipelined requests are received
302         //     and they are handled by different threads simultaneously.
303         //     e.g. when the 3rd request triggers a reset and then the 2nd one triggers another.
304         minClosedId = Math.min(minClosedId, id);
305 
306         if (minClosedId <= maxIdWithPendingWrites) {
307             final ClosedSessionException cause =
308                     new ClosedSessionException("An HTTP/1 stream has been reset: " + error);
309             for (int i = minClosedId; i <= maxIdWithPendingWrites; i++) {
310                 final PendingWrites pendingWrites = pendingWritesMap.remove(i);
311                 for (;;) {
312                     final Entry<HttpObject, ChannelPromise> e = pendingWrites.poll();
313                     if (e == null) {
314                         break;
315                     }
316 
317                     e.getValue().tryFailure(cause);
318                 }
319             }
320         }
321 
322         final ChannelFuture f = ch.write(Unpooled.EMPTY_BUFFER);
323         if (!isWritable(currentId)) {
324             f.addListener(ChannelFutureListener.CLOSE);
325         }
326 
327         return f;
328     }
329 
330     @Override
331     public final boolean isWritable(int id, int streamId) {
332         return isWritable(id);
333     }
334 
335     protected final boolean isWritable(int id) {
336         return id < minClosedId;
337     }
338 
339     @Override
340     public final void close() {
341         if (closed) {
342             return;
343         }
344         closed = true;
345 
346         if (pendingWritesMap.isEmpty()) {
347             return;
348         }
349 
350         final ClosedSessionException cause = ClosedSessionException.get();
351         for (Queue<Entry<HttpObject, ChannelPromise>> queue : pendingWritesMap.values()) {
352             for (;;) {
353                 final Entry<HttpObject, ChannelPromise> e = queue.poll();
354                 if (e == null) {
355                     break;
356                 }
357 
358                 e.getValue().tryFailure(cause);
359             }
360         }
361 
362         pendingWritesMap.clear();
363     }
364 
365     @Override
366     public boolean isClosed() {
367         return closed;
368     }

370     private static final class PendingWrites extends ArrayDeque<Entry<HttpObject, ChannelPromise>> {
371 
372         private static final long serialVersionUID = 4241891747461017445L;
373 
374         private boolean endOfStream;
375 
376         PendingWrites() {
377             super(4);
378         }
379 
380         @Override
381         public boolean add(Entry<HttpObject, ChannelPromise> httpObjectChannelPromiseEntry) {
382             return isEndOfStream() ? false : super.add(httpObjectChannelPromiseEntry);
383         }
384 
385         boolean isEndOfStream() {
386             return endOfStream;
387         }
388 
389         void setEndOfStream() {
390             endOfStream = true;
391         }
392     }

394     protected final SessionProtocol protocol() {
395         return protocol;
396     }
